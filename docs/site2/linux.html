# Permission examples: chmod 755 script.sh # rwxr-xr-x chmod +x script.sh # Add execute permission chmod -R 755 folder/ # Recursive permissions

Package Management

APT (Debian/Ubuntu)

sudo apt update # Update package list sudo apt upgrade # Upgrade packages sudo apt install [pkg] # Install package sudo apt remove [pkg] # Remove package sudo apt search [term] # Search packages

YUM/DNF (RHEL/CentOS/Fedora)

sudo dnf update # Update packages sudo dnf install [pkg] # Install package sudo dnf remove [pkg] # Remove package

Service Management

# Systemd (most modern distros) sudo systemctl start nginx # Start service sudo systemctl stop nginx # Stop service sudo systemctl restart nginx # Restart service sudo systemctl status nginx # Check status sudo systemctl enable nginx # Enable on boot sudo systemctl disable nginx # Disable on boot
